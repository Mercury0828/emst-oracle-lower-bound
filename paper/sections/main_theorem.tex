\section{Estimating the MST weight of \texorpdfstring{$P$}{P}}\label{sec:main}

We now assemble the pieces of \cref{sec:dt} into the main result.

\begin{theorem}[Main]\label{thm:main}
For every fixed $\eps\in(0,1)$ and every $P\subseteq[\Delta]^2$ with $\abs P=n$ and $\Delta=O(n)$, there
is a randomized algorithm that, with probability at least $2/3$, outputs $\widehat W$ with
\[
  (1-\eps)\,\wt(\MST(P))\ \le\ \widehat W\ \le\ (1+\eps)\,\wt(\MST(P)),
\]
using $\Otil_\eps\!\bigl(n^{1/3}\bigr)$ range-counting queries to $P$.
\end{theorem}

\begin{algorithm}[t]
\caption{Estimating $W=\wt(\MST(P))$ to a $(1\pm\eps)$ factor}\label{alg:main}
\KwIn{oracle access to $P\subseteq[\Delta]^2$; accuracy $\eps$}
\KwOut{$\widehat W$ with $(1-\eps)W\le\widehat W\le(1+\eps)W$ w.p.\ $\ge 2/3$}
$U\leftarrow\sqrt2(n-1)\Delta$; $\;\zeta_0\leftarrow\tfrac1{64}$\;
\ForEach{$G=U,U/2,U/4,\dots$ \textnormal{(at most $O(\log n)$ guesses)}}{
  $\widehat W(G)\leftarrow$ median of $O(\log n)$ runs of \textsc{Estimate}$(G,\zeta_0)$ \tcp*{aborted run $=+\infty$}
  \lIf{$\widehat W(G)\ge G/8$}{$G_0\leftarrow G$; \textbf{break}}
}
\Return median of $O(\log n)$ runs of \textsc{Estimate}$(G_0,\eps/32)$\;
\medskip
\textbf{procedure} \textsc{Estimate}$(G,\zeta)$, with $\xi=\zeta/C_*$, $K_0=\ceil{n^{2/3}}$, $L=G/K_0$:\;
\Indp
regularize to a support $Q$ with $K=\Theta_\xi(K_0)$ cells (\cref{lem:kwindow}); estimate $K$\;
$\widehat A_P\leftarrow$ clipped death-time sampler on a $(1{+}\rho)$-spanner of $P$, $s_P=O(\xi^{-2}n^{1/3})$ seeds\;
$\widehat\Wq\leftarrow$ support-MST estimator (\cref{lem:support}) at relative accuracy $\Theta(\xi)$\;
$\widehat A_Q\leftarrow$ clipped death-time sampler on a $(1{+}\rho)$-spanner of $Q$, $s_Q=O_\xi(1)$ seeds\;
\Return $\widehat A_P+\widehat\Wq-\widehat A_Q$ (or \textsc{fail} if the global query budget is exceeded)\;
\end{algorithm}

\subsection{Estimating the three pieces under a guess}\label{sec:main-pieces}

Fix a guess $G\ge W$ and an internal accuracy $\xi$, and set $K_0=\ceil{n^{2/3}}$, $L=G/K_0$, with the
regularized support $Q$ of \cref{sec:dt-regularize} ($K=\Theta_\xi(K_0)$). The decomposition
\eqref{eq:decomp} expresses $W$ through three estimable quantities, $A_L(P)$, $\Wq$, and $A_L(Q)$. We
estimate each to additive error $O(\xi G)$; the common normalization by $G$ (not by the individual
quantities) is what lets the same internal accuracy serve all three and the geometric search below. The
clip level $L=G/K_0$ is the lever: it is small enough that the death-time variance \eqref{eq:dt-var},
which scales with $L$, stays affordable, and large enough that the tail beyond $L$ carries only the heavy
edges the support handles. Each estimate:
\begin{itemize}[leftmargin=1.5em,itemsep=2pt]
\item \emph{Bulk $A_L(P)$.} Run the death-time sampler on a $(1+\rho)$-spanner of $P$ with
  $\rho=\Theta(\xi)$. By \eqref{eq:dt-var} with $\abs V=n$, $L=G/K_0$ and $\wt(\MST(P))=W\le G$, the
  variance is $O(nG^2/(K_0 s_P))$, so $s_P=O(\xi^{-2}n/K_0)=O(\xi^{-2}n^{1/3})$ samples suffice; with
  $\Otil(1)$ queries per sample this is $\Otil_\xi(n^{1/3})$ queries (the spanner bias is
  $\le\rho W=O(\xi G)$ by \eqref{eq:spanner-stretch}).
\item \emph{Support weight $\Wq$.} Apply \cref{lem:support} with relative error $\Theta(\xi/C_W)$; since
  $\Wq\le C_W W\le C_W G$ this gives additive error $O(\xi G)$ in $\Otil_\xi(\sqrt K+n/K)=\Otil_\xi(n^{1/3})$
  queries.
\item \emph{Support bulk $A_L(Q)$.} Run the death-time sampler on a $(1+\rho)$-spanner of $Q$. Here
  $\abs V=K=\Theta(K_0)$, $L=G/K_0$ and $\wt(\MST(Q))=\Wq=O(G)$, so \eqref{eq:dt-var} makes the variance
  $O(G^2/s_Q)$ and $s_Q=O_\xi(1)$ support samples suffice; each uniform support sample costs $\Otil(n/K)
  =\Otil_\xi(n^{1/3})$ queries (\cref{sec:support-oracle}).
\end{itemize}
No sample seed is reused across thresholds or across the three estimators, so the three errors are
independent.

\begin{remark}[The choice $K_0=n^{2/3}$]\label{rem:k0}
The support size $K_0$ is the only free knob, and the three costs above pin it down. The support estimator
costs $\Otil(\sqrt{K_0}+n/K_0)$; the bulk sampler costs $\Otil(\xi^{-2}n/K_0)$ (more support cells mean a
smaller clip $L=G/K_0$, hence lower death-time variance and fewer seeds); the support sampler costs
$\Otil(n/K_0)$ per seed. The first is increasing in $K_0$ through its $\sqrt{K_0}$ term and the rest are
decreasing, and all four meet when $\sqrt{K_0}=\Theta(n/K_0)$, i.e.\ $K_0=\Theta(n^{2/3})$, giving the
common value $\Otil(n^{1/3})$. Any other choice makes one of the terms dominate and the bound worse.
\end{remark}

\subsection{Combining and removing the guess}\label{sec:main-combine}

Output
\begin{equation}\label{eq:combine}
  \widehat W(G)=\widehat A_P+\widehat\Wq-\widehat A_Q .
\end{equation}
By the decomposition \eqref{eq:decomp} and the tail transfer \eqref{eq:tail-transfer},
\[
  \abs{\widehat W(G)-W}\le\abs{\widehat A_P-A_L(P)}+\abs{\widehat\Wq-\Wq}+\abs{\widehat A_Q-A_L(Q)}
    +\abs{B_L(Q)-B_L(P)},
\]
and each of the four terms is $O(\xi G)$, so for a suitable constant $C_*$,
\begin{equation}\label{eq:guess-accuracy}
  \abs{\widehat W(G)-W}\le C_*\xi G\qquad\text{whenever }G\ge W .
\end{equation}
Rescaling the internal accuracy to $\zeta=C_*\xi$ gives $\abs{\widehat W(G)-W}\le\zeta G$ in
$\Otil_\zeta(n^{1/3})$ queries per call. A single run of the call carries the deterministic budget
$B=\Otil_\zeta(n^{1/3})$ of \cref{sec:wspd-cap} and returns \textsc{fail} if it aborts. We amplify each
guess into a robust estimate by taking the median of $t=O(\log n)$ independent runs, where an aborted run
contributes $+\infty$ to the median. For a guess with $G\ge W$, one run is \emph{bad} if it aborts
(probability $\le 1/C$) or is inaccurate (probability $\le 1/6$); choosing the budget constant $C$ large
makes the bad probability less than $1/3$, so fewer than half of the $t$ runs are bad with probability
$1-n^{-\Omega(1)}$, and the median is then an accurate, finite estimate. A union bound over the $O(\log
n)$ guesses keeps all guesses with $G\ge W$ simultaneously accurate with constant probability; guesses
with $G<W$ need no guarantee beyond not stopping the search prematurely, which the accurate guesses above
already enforce.

The estimator needs a guess $G\ge W$ to set its clip level $L=G/K_0$ and its budgets, but $W$ is what we
are computing. The way out is that the guarantee \eqref{eq:guess-accuracy} is one-sided in a useful way:
for any $G\ge W$ the output $\widehat W(G)$ is within $\zeta G$ of $W$, so reading $\widehat W(G)$ against
the scale of $G$ tells us whether $G$ is still far above $W$ or has come down near it. Sweeping $G$
downward by halving and watching for $\widehat W(G)$ to first reach a constant fraction of $G$ therefore
brackets $W$ within a factor of two, after which one calibrated call finishes the job. We make this
precise.

A deterministic upper bound is $U=\sqrt2(n-1)\Delta\ge W$, and $W\ge n-1$ since the points are distinct
integer-grid points, so $U/W=O(\Delta)=O(n)$ and only $O(\log n)$ guesses are needed. Run the per-guess
estimator with $\zeta_0=\tfrac1{64}$ on $G=U,U/2,U/4,\dots$, amplified so that with constant probability
every call with $G\ge W$ is correct, and stop at the first $G$ with $\widehat W(G)\ge G/8$. If $G\ge16W$
then $\widehat W(G)\le W+G/64\le G/16+G/64<G/8$, so the search does not stop; if $G\in[W,2W)$ then
$\widehat W(G)\ge W-G/64>G/8$, so it stops while $G\ge W$. Hence the selected $G_0$ satisfies $W\le
G_0<16W$. A final call on $G_0$ with $\zeta=\eps/32$, amplified by the same median of $O(\log n)$ runs,
gives $\abs{\widehat W-W}\le(\eps/32)G_0<(\eps/2)W$. Allotting each of the $O(\log n)$ guess calls and the
final call a failure budget $O(1/\log n)$ keeps the whole procedure correct with probability $\ge 2/3$,
and the $O(\log n)$ guesses keep the total cost $\Otil_\eps(n^{1/3})$. This proves \cref{thm:main}; the
deferred variance, packing, and snapping calculations are completed in \cref{sec:app-assembly}.

\begin{remark}[Running time]\label{rem:time}
Although \cref{thm:main} counts oracle queries, the algorithm is also efficient in the usual sense. Every
step between queries is bookkeeping on cells and ranks: the cover construction, the offline candidate
generation, the leader explorations, the WSPD partner enumeration, and the bottleneck priority queue each
do work proportional to the number of cells touched, which is within a polylogarithmic factor of the query
count. On a word RAM with $O(\log n)$-bit words, where a quadtree cell, a representative's integer coordinates,
and a rank fit in $O(1)$ words, the total running time is therefore $\Otil_\eps(n^{1/3})$ as well,
dominated by the queries up to polylogarithmic factors. The only numeric operations are comparisons and
additions of edge lengths in the bottleneck search and the integration: edge-length comparisons are done
on \emph{squared} Euclidean distances, which are exact integers, and the clipped-weight sums need each
length only to relative precision $\poly(\eps/\log n)$, supplied by $O(\log(n/\eps))$-bit approximations of
the square roots. So no exact irrational arithmetic is required, and the algorithm runs in the standard
integer word-RAM model.
\end{remark}

\subsection{The algorithm on a concrete instance}\label{sec:main-example}

To see the pieces act in concert, follow them on the hard shape of \cref{sec:overview}, with parameters
that stay inside the model $\Delta=O(n)$. The input is a unit-spaced background carpet of $n-s$ points
filling a $\Theta(\sqrt n)\times\Theta(\sqrt n)$ region, together with only $s=\Theta(\log n)$
\emph{satellite} points placed so that the MST links them by a chain of geometrically growing edges of
lengths $2^1,2^2,\dots,2^s$, the longest reaching $2^s=\Theta(\Delta)=\Theta(n)$. The carpet contributes
$\Theta(n)$ to the MST weight (each point joins a neighbor at distance $\Theta(1)$) and the chain
contributes $\Theta(\sum_{i\le s}2^i)=\Theta(2^s)=\Theta(n)$, so $W=\Theta(n)$, with a constant fraction of
it living in the few long satellite edges.

The geometric search calibrates a guess $G_0\in[W,16W)$, fixing $L=G_0/K_0=\Theta(n^{1/3})$ with
$K_0=\Theta(n^{2/3})$. Regularization (\cref{lem:kwindow}) starts at scale $\Theta(\xi L)$ and halves to
the first scale with $\Theta(K_0)$ occupied cells; for this carpet that scale is $\Theta(n^{1/6})$, at
which the carpet collapses to $\Theta(K_0)$ cells and each of the $s$ far-apart satellites still occupies
its own cell. The point of the instance is the contrast at the support: the satellites are a vanishing
fraction $s/K_0=\Theta(\log n/n^{2/3})$ of the support cells, so a sample of $\poly\log$ \emph{support
points} would miss them, yet the long satellite edges carry a constant fraction of $\Wq$. The empty-cell
leader estimator samples cells, hitting each satellite cell with probability $\Theta(1/K_0)$, so it reads
the full $\Wq=\Theta(n)$, heavy tail included, in $\Otil(\sqrt{K_0})=\Otil(n^{1/3})$ queries.

The three estimators then combine through $W=A_L(P)+B_L(P)$ with $B_L(P)\approx B_L(Q)=\Wq-A_L(Q)$. The
death-time samplers estimate the two clipped bulks $A_L(P)$ and $A_L(Q)$ directly (each dominated by the
short edges its set carries), and $\widehat\Wq$ supplies the heavy tail; in the assembly $\widehat W=
\widehat A_P+\widehat\Wq-\widehat A_Q$ the support's own bulk $A_L(Q)$ is subtracted off and the satellite
tail survives, returning $\Theta(n)=\Theta(W)$. A purely point-mass algorithm at the same budget never
samples a satellite and reports only the carpet weight, missing the constant-fraction tail; the spatial
leader estimator is exactly what averts that failure.
