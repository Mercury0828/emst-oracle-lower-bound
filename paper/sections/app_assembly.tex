\section{Full proofs for the death-time primitive and the reduction}\label{sec:app-assembly}

\subsection{The death-time sampler}\label{sec:app-dt}

\begin{proof}[Proof of \eqref{eq:death-multiset} and \eqref{eq:dt-var}]
Fix a rank assignment and a threshold $t$, and let $G_H(t)$ be the graph on $V$ with the edges of $H$ of
weight $\le t$. A vertex $v$ has $\tau(v)>t$ iff every path in $H$ from $v$ to the root or to a
lower-ranked vertex uses an edge of weight $>t$; equivalently the component of $v$ in $G_H(t)$ contains
neither the root nor any vertex of smaller rank, i.e.\ $v$ is the minimum-rank vertex of a component of
$G_H(t)$ that does not contain the root. Each component of $G_H(t)$ has exactly one minimum-rank vertex,
and one of these components is the root's, so
\[
  \abs{\{v:\tau(v)>t\}}=(\#\text{components of }G_H(t))-1=c_H(t)-1,
\]
which by \eqref{eq:components-edges} equals the number of MST edges of weight $>t$. Two nonnegative
integer-valued measures on $[0,\infty)$ with equal right tails at every $t$ are equal, so the multiset
$\{\tau(v)\}$ equals $\{0\}\cup\{\wt(e):e\in\MST(H)\}$ (the lone $0$ being the root's death time); this is
\eqref{eq:death-multiset}.

For the clipped statistic, integrate the tail identity against the clip kernel. Since $\min\{\tau(v),L\}
=\int_0^L\mathbf 1[\tau(v)>t]\,dt$,
\[
  \sum_{v\in V}\min\{\tau(v),L\}=\int_0^L\abs{\{v:\tau(v)>t\}}\,dt=\int_0^L(c_H(t)-1)\,dt
    =\sum_{e\in\MST(H)}\min\{\wt(e),L\}=A_L(H),
\]
so for a uniformly random $v$ we have $\E X_L=A_L(H)/\abs V$, i.e.\ $\abs V\cdot\E X_L=A_L(H)$, and the
sample mean scaled by $\abs V$ is unbiased for $A_L(H)$. For the variance, $0\le X_L\le L$ gives
$X_L^2\le L\,X_L$, hence
\[
  \Var(X_L)\le\E X_L^2\le L\,\E X_L=\frac{L\,A_L(H)}{\abs V}\le\frac{L\,\wt(\MST(H))}{\abs V}.
\]
Averaging $s$ independent samples divides the variance by $s$; scaling by $\abs V$ multiplies it by
$\abs V^2$, giving $\Var[\tfrac{\abs V}{s}\sum_i X_L(v_i)]\le\abs V\,L\,\wt(\MST(H))/s$, which is
\eqref{eq:dt-var}.
\end{proof}

\begin{proof}[Proof of \eqref{eq:spanner-stretch}]
$H_X$ is a Euclidean graph on $X$, so $c_{H_X}(t)\ge c_X(t)$ (it has at most the threshold-$t$ edges of the
complete graph); and it is a $(1+\rho)$-spanner, so an edge of length $\le t/(1+\rho)$ in the complete
graph is matched by an $H_X$-path of bottleneck $\le t$, giving $c_{H_X}(t)\le c_X(t/(1+\rho))$.
Integrating $c-1$ against the clip kernel $\mathbf 1[t\le L]$ and substituting yields
$A_L(X)\le A_L(H_X)\le(1+\rho)A_L(X)$, where the right factor absorbs the $t\mapsto(1+\rho)t$ rescaling and
$A_L\le\wt(\MST(X))$.
\end{proof}

\subsection{Support regularization and snapping}\label{sec:app-snap}

\begin{proof}[Proof of \eqref{eq:Kh-packing}]
Choose one point of $P$ in each nonempty $h$-cell and four-color the cells by index parities. In each
color class the chosen points are pairwise at distance $\ge h$, so the class has $\le 1+\wt(\MST(\text{class}))/h
\le 1+2W/h$ points by \eqref{eq:doubling} applied to $P$. Summing the four classes,
$K_h\le 4+8W/h$.
\end{proof}

\begin{lemma}[The regularization search]\label{lem:kwindow}
Fix a small constant $c\in(0,1)$ and set $C_\xi=\Theta(1/\xi)$ with $C_\xi/c>16$. Given a guess $G\ge W$,
start at $h_0=\Theta(\xi L)$ and, if the nonempty-cell count $K_{h_0}$ is certified below $cK_0$, halve $h$
until it is first certified at least $cK_0$. The resulting scale $h$ satisfies $h\le h_0=\Theta(\xi L)$ and
$\tfrac{c}{4}K_0\le K=K_h\le C_\xi K_0$, using $\Otil_\xi(n/K_0)=\Otil(n^{1/3})$ range-counting queries in
total.
\end{lemma}

\begin{proof}
As $h$ halves, each nonempty cell splits into at most four, so $K_h$ is nondecreasing as $h$ decreases and
grows by a factor at most $4$ per halving. We only ever halve, so the final scale obeys $h\le
h_0=\Theta(\xi L)$, which is what the snapping bound $\delta_s=\sqrt2\,h=O(\xi L)$ of \cref{sec:dt-snap}
needs.

\emph{Upper bound on $K$.} At the initial scale, $K_{h_0}\le 8W/h_0+4=O(K_0/\xi)\le C_\xi K_0$ by
\eqref{eq:Kh-packing} and $h_0=\Theta(\xi G/K_0)\ge\Theta(\xi W/K_0)$. If $K_{h_0}\ge cK_0$ we stop at
$h_0$ and $K=K_{h_0}\in[cK_0,C_\xi K_0]$. Otherwise let $h$ be the first halved scale certified at least
$cK_0$; the previous scale was below $cK_0$, so the quadrupling bound gives $K<4cK_0\le C_\xi K_0$. (A
finest scale always crosses, since at cell side $1$ every point is its own cell and $K=n\ge K_0$.) Either
way $K\le C_\xi K_0$.

\emph{Lower bound on $K$ and the test.} At each scale we run a fixed batch of $m=\Theta(\xi^{-2}(n/K_0)\log
n)$ rejection trials (\cref{sec:support-oracle}): a trial draws a uniform point of $P$ ($O(\log\Delta)$
counts, \cref{lem:telescoping}) and accepts with probability $1/\abs{P\cap C}$, so it accepts with
probability exactly $K_h/n$. A Chernoff bound on the accept count separates $K_h\ge cK_0$ from
$K_h<\tfrac{c}{4}K_0$ with failure probability $n^{-\Omega(1)}$, so the threshold test ``accepts
$\ge\tfrac{c}{2}K_0\,m/n$'' is correct at every scale with high probability. We stop at the first scale the
test accepts; there $K_h\ge\tfrac{c}{4}K_0$ (else the test would have rejected). Combined with the upper
bound, $K\in[\tfrac{c}{4}K_0,C_\xi K_0]$, i.e.\ $\Theta_\xi(K_0)$, which is \eqref{eq:K-window} with the
constants relabeled.

\emph{Cost.} The test spends $\Otil(m)=\Otil_\xi(n/K_0)$ counts per scale regardless of how small $K_h$
is, and runs at $O(\log\Delta)$ scales, for $\Otil_\xi(n/K_0)=\Otil(n^{1/3})$ total. Once stopped, the
same batch estimates $K$ to relative error $O(\xi)$ (its accept count concentrates at $K_h m/n$), the
input to \cref{lem:support} required by \cref{sec:main-pieces}. Both the support estimator
$\Otil_\eta(\sqrt K+n/K)$ and the support death-time sampler are run at this $K=\Theta_\xi(K_0)$, and
since $\sqrt{C_\xi K_0}=\Otil_\xi(n^{1/3})$ and $n/(\tfrac{c}{4}K_0)=\Otil_\xi(n^{1/3})$, all three pieces
of \cref{sec:main-pieces} remain $\Otil_\xi(n^{1/3})$.
\end{proof}

\begin{proof}[Proof of \eqref{eq:snap-interleave}--\eqref{eq:tail-transfer} and $\Wq\le C_W W$]
Snapping moves each point by at most $\delta_s=\sqrt2\,h$. If two points are within $t$ before snapping,
their centers are within $t+\delta_s$; if two centers are within $t$, the original points are within
$t+\delta_s$. Hence a connected component at threshold $t$ for one set is connected at threshold
$t+\delta_s$ for the other, giving $c_P(t+\delta_s)\le\cQ(t)\le c_P(t-\delta_s)$, which is
\eqref{eq:snap-interleave}. Integrating over $[L,\infty)$ and using
$B_L(X)=\int_L^\infty(c_X-1)=\sum_e(\abs e-L)_+$,
\[
  B_{L+\delta_s}(P)\le B_L(Q)\le B_{L-\delta_s}(P),
\]
so $\abs{B_L(Q)-B_L(P)}\le\int_{L-\delta_s}^{L+\delta_s}(c_P(t)-1)\,dt\le\int_{L-\delta_s}^{L+\delta_s}
\frac Wt\,dt\le\frac{2\delta_s}{L-\delta_s}W\le\frac{3\delta_s}{L}W$ for $\delta_s\le L/2$, which is
\eqref{eq:tail-transfer} since $\delta_s=\sqrt2\,h=O(\xi L)$ (\cref{lem:kwindow}). For the last bound, contract $\MST(P)$
by the $h$-cells and keep a spanning tree of the resulting graph on $Q$; each retained edge grows by at
most $\delta_s$ under snapping, so $\Wq\le W+\delta_s(K-1)$. By \eqref{eq:Kh-packing} and $K=\Omega(K_0)$,
$\delta_s K=O(hW/h)=O(W)$, giving $\Wq\le C_W W$.
\end{proof}

\subsection{The assembly bound (\texorpdfstring{\eqref{eq:guess-accuracy}}{guess accuracy})}\label{sec:app-combine}

\begin{proof}[Proof of \eqref{eq:guess-accuracy}]
Assume $G\ge W$, so $L=G/K_0\ge W/K_0$ and $\delta_s=O(\xi L)$ satisfies $\delta_s\le L/2$ for small
$\xi$. We give each of the four contributions to \eqref{eq:combine}--\eqref{eq:decomp} a budget of
$\tfrac{C_*}{4}\xi G$ and verify each in turn; set $\rho=\Theta(\xi)$ and recall $W\le G$ and $\Wq\le C_W
W\le C_W G$.

\emph{(i) Bulk of $P$, $\widehat A_P$.} Two errors enter. The spanner bias is one-sided and at most $\rho
W$ by \eqref{eq:spanner-stretch}, which is $\le\tfrac{C_*}{12}\xi G$ for $\rho$ a small enough multiple of
$\xi$. The sampling error: by \eqref{eq:dt-var} with $\abs V=n$, $L=G/K_0$, and $\wt(\MST(H_P))\le(1+\rho)W
\le 2G$, the variance of the $s_P$-sample mean (scaled by $n$) is at most $nL\cdot 2G/s_P=2nG^2/(K_0 s_P)$;
with $s_P=C\xi^{-2}n/K_0$ this is $\tfrac{2}{C}\xi^2G^2$, so Chebyshev gives deviation $\le\tfrac{C_*}{12}
\xi G$ with probability $\ge 1-\tfrac1{100}$ for $C$ large. Hence $\abs{\widehat A_P-A_L(P)}\le\tfrac{C_*}
4\xi G$.

\emph{(ii) Support weight, $\widehat\Wq$.} \Cref{lem:support} with relative accuracy $\eta=\xi/(4C_WC_*^{-1})
=\Theta(\xi)$ gives $\abs{\widehat\Wq-\Wq}\le\eta\Wq\le\eta C_W G\le\tfrac{C_*}4\xi G$.

\emph{(iii) Bulk of $Q$, $\widehat A_Q$.} Identical to (i) with $\abs V=K=\Theta(K_0)$ and $\wt(\MST(H_Q))
\le(1+\rho)\Wq=O(G)$: the variance is $O(K L G/s_Q)=O(G^2/s_Q)$, so $s_Q=O(\xi^{-2})$ samples give
$\abs{\widehat A_Q-A_L(Q)}\le\tfrac{C_*}4\xi G$ with probability $\ge 1-\tfrac1{100}$.

\emph{(iv) Tail transfer.} $\abs{B_L(Q)-B_L(P)}\le\tfrac{C_*}4\xi G$ is \eqref{eq:tail-transfer}, which
gave $O(\xi W)$.

Summing the four through \eqref{eq:decomp} (note $W=A_L(P)+B_L(P)$ and $B_L(Q)=\Wq-A_L(Q)$),
\[
  \abs{\widehat W(G)-W}=\bigl|(\widehat A_P-A_L(P))+(\widehat\Wq-\Wq)-(\widehat A_Q-A_L(Q))
    +(B_L(P)-B_L(Q))\bigr|\le C_*\xi G,
\]
which is \eqref{eq:guess-accuracy}. Each piece fails with probability $\le\tfrac1{50}$, so one call is
within $C_*\xi G$ with probability $\ge\tfrac{9}{10}$; the median of $O(\log n)$ calls (\cref{sec:main-combine})
then makes every guess with $G\ge W$ correct simultaneously with constant probability, an aborted call
counting as a failed one.
\end{proof}

The cost of one per-guess call is $\Otil_\xi(n^{1/3})$ (the maximum of the three pieces in
\cref{sec:main-pieces}), and the $O(\log n)$-guess search multiplies this by $O(\log n)$, so
\cref{thm:main} runs in $\Otil_\eps(n^{1/3})$ range-counting queries. \qed
